cmake_minimum_required(VERSION 3.14)
project(schnitzel NONE)

include(ExternalProject)

set(SCHNITZEL_ROOT ${CMAKE_CURRENT_SOURCE_DIR})

ExternalProject_Add(sim
    SOURCE_DIR  ${SCHNITZEL_ROOT}/sim
    BINARY_DIR  ${CMAKE_CURRENT_BINARY_DIR}/sim
    CMAKE_ARGS  -DSCHNITZEL_ROOT=${SCHNITZEL_ROOT}
                -DCMAKE_BUILD_TYPE=${CMAKE_BUILD_TYPE}
    BUILD_ALWAYS ON
    INSTALL_COMMAND ""
)

ExternalProject_Add(sw_device
    SOURCE_DIR  ${SCHNITZEL_ROOT}/sw/device
    BINARY_DIR  ${CMAKE_CURRENT_BINARY_DIR}/sw/device
    CMAKE_ARGS  -DSCHNITZEL_ROOT=${SCHNITZEL_ROOT}
                -DCMAKE_BUILD_TYPE=${CMAKE_BUILD_TYPE}
    BUILD_ALWAYS ON
    INSTALL_COMMAND ""
)

ExternalProject_Add(sw_host
    SOURCE_DIR  ${SCHNITZEL_ROOT}/sw/host
    BINARY_DIR  ${CMAKE_CURRENT_BINARY_DIR}/sw/host
    CMAKE_ARGS  -DSCHNITZEL_ROOT=${SCHNITZEL_ROOT}
                -DCMAKE_BUILD_TYPE=${CMAKE_BUILD_TYPE}
    BUILD_ALWAYS ON
    INSTALL_COMMAND ""
)
